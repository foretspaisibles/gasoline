Module Unicode.UBuffer

module UBuffer: sig .. end
Buffers containing unicode characters
type t 
val create : int -> t
create n creates the buffer which initially can contain n Unicode characters.
val contents : t -> Unicode.ustring
contents b returns a copy of the current content of the buffer. The buffer itself is unchanged.
val sub : t -> int -> int -> Unicode.ustring
sub b off len returns a copy of the substring of the current content of the buffer, starting at position off and for len characters.
Raises Invalid_argument if out of bounds request.
val nth : t -> int -> Unicode.uchar
nth b n gets the n-th character in buffer b.
Raises Invalid_argument if out of bounds request.
val length : t -> int
length b returns the number of characters currently contained in the buffer.
val clear : t -> unit
val reset : t -> unit
val add_char : t -> char -> unit
val add_string : t -> string -> unit
val add_uchar : t -> Unicode.uchar -> unit
val add_ustring : t -> Unicode.ustring -> unit
val add_buffer : t -> t -> unit
val output_buffer : Unicode.UChannel.out_channel -> t -> unit
val join : t -> Unicode.ustring -> Unicode.ustring list -> unit
join b sep l adds to b the elements of l, separated by sep.
