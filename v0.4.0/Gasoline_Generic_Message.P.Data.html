Module Gasoline_Generic_Message.P.Data

module Data: Gasoline_Generic_Definition.DATA 
type word
The type of data words.
type block
The type of data blocks.
type out_channel
The type of data output channels.
val output_word : out_channel ->
       word -> unit
Write a word on the given output channel.
val output_block : out_channel ->
       block -> int -> int -> unit
output oc buf pos len write len words from block buf, starting at offset pos, to the given output channel oc.
Raises Invalid_argument "output" if pos and len do not designate a valid substring of buf.
val flush : out_channel -> unit
Flush an output channel, to empty its internal buffer.
